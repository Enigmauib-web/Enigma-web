import styles from './utils.module.css';

export const Bilde = ({children, url, alt}) => {
    return (
        <div className={styles.img}>
            <img src={url} alt={alt}/>
            <p className={styles.alt}>{alt}</p>
        </div>
    )
}

export const FolkBilder = ({children, name, title, img}) => {
    return (
        <div className={styles.leder}>
            <img src={img} className={styles.circle}></img>
            <h3>{name}</h3>
            <h4>{title}</h4>
        </div>
    )
}

export const FolkDiv = ({children, folk}) => {
    return (
        <div className={styles.lederDiv} >
            { folk.map( (props, idx) => (<FolkBilder key={idx} {...props}/>) ) }
        </div>
    )
}

export const Header = ({children}) => {
    return (
        <h1 className={styles.header}>{children}</h1>
    )
}

export const BildeDiv = ({children, urls}) => {
    return(
        <div className={styles.bildediv}>
            {urls.map(url => <Bilde url={url} />)}
        </div>
    )
}

export const Vimeo = ({ video, title }) => {
  const vimeoSource = `https://player.vimeo.com/video/${video}`;
  return (
    <div style={{
      paddingTop: "56.25%",
      position: "relative",
    }}>
      <iframe
        src={vimeoSource}
        style={{
          position: "absolute",
          top: 0,
          left: 0,
          width: "100%",
          height: "100%",
        }}
        title={title}
        frameBorder="0"
        allow="autoplay; fullscreen; picture-in-picture"
        data-ready="true"
      />
    </div>
  );
};

export const Player = ({videoSource, title}) => {
    return (
        <div style={{
        paddingTop: "56.25%",
        position: "relative",
        }}>
        <iframe
            sandbox=''
            src={videoSource}
            style={{
            position: "absolute",
            top: 0,
            left: 0,
            width: "auto",
            height: "100%",
            }}
            title={title}
            frameBorder="0"
            allow="fullscreen; picture-in-picture"
            data-ready="true"
        />
        </div>
    )
}
